class Node {
    int getIntValue(string name, int default = 0);
    string getStringValue(string name, string default = "");
    //... ebenso für z.B. bool

    void setIntValue(string name, int value);
    void setStringValue(string name, string value);
    //... ebenso für z.B. bool

    Node* getSubNodeIndexed(int index);
